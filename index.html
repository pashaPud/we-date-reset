  <h1>Сброс пароля</h1>

  <form id="resetForm">

    <input type="password" id="password" placeholder="Новый пароль" required />

    <input type="password" id="confirm" placeholder="Повторите пароль" required />

    <button type="submit">Установить новый пароль</button>

    <div class="message" id="message"></div>

  </form>


  <script type="module">

    import { createClient } from 'https://cdn.jsdelivr.net/npm/@supabase/supabase-js/+esm'


    const supabaseUrl = 'https://tfewdarpiwdelivlifuu.supabase.co'

    const supabase = createClient(supabaseUrl)


    const form = document.getElementById('resetForm')

    const message = document.getElementById('message')


    form.addEventListener('submit', async (e) => {

      e.preventDefault()


      const password = document.getElementById('password').value

      const confirm = document.getElementById('confirm').value


      if (password !== confirm) {

        message.textContent = 'Пароли не совпадают'

        return

      }


      const { error } = await supabase.auth.updateUser({ password })


      if (error) {

        message.textContent = 'Ошибка: ' + error.message

      } else {

        message.textContent = 'Пароль успешно изменён! Можете закрыть окно.'

        form.reset()

      }

    })

  </script>
